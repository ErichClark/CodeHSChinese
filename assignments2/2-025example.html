Example Code 示例代碼

// This program will draw an 8 ball   //此程序將繪製一個8球
// centered on the screen. The circle   //在屏幕上居中。圓
// should be black with radius 100, and   //應該是黑色的，半徑100，和
// the "8" should be white in 50pt Arial   //“8”應該是白色的50點宋體
// font.   //字體。
function start(){   函數的開始（）{
   var ball = new Circle(100);   可變球=新圓（100）;
   var centerX = getWidth() / 2;   可變中心X =得到寬度（）/2;
   var centerY = getHeight() / 2;   可變中心Y = GET高度（）/2;
   ball.setPosition(centerX, centerY);   球。設定位置（中心X，中心Y）;
   add(ball);   添加（球）;

   var number = new Text("8");   可變數目=新文本（“8”）;
   number.setFont("50pt Arial");   數。設置字體（“50點宋體”）;
   number.setColor(Color.white);   數。設置顏色（顏色。白）;
   number.setPosition(centerX - number.getWidth()/2, centerY + number.getWidth()/2);   數。設定位置（。中心x - 數得到寬（）/2，中心Y +數量得到寬度（）/2）;
   add(number);   添加（數量）;
}
