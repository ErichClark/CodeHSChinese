var MAX = 100;   變量最大值=100;

function start(){   函數的開始（）{
   var sum = 0;   變量總和=0;
   for(var i = MIN; i <= MAX; i++){   對於（可變 i = 最低限度; i <= 最大; i++){
     println(sum += i);  
   }

   println("The sum was " + sum);   打印線（“和睦是”+總和）;
}
