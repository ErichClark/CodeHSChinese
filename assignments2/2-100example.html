function start(){
	var numApples = 4;
	println("Before: " + numApples);
	var twiceAsMany = doubleNumber(numApples);
	println("After: " + twiceAsMany);
	
	var y = 11;
	println(doubleNumber(y));
}

function doubleNumber(x){
	var doubledX = 2 * x;
	return doubledX;
}

函數的開始（）{
	可變數量的蘋果= 4;
	打印線(“前：” + 蘋果的數 );
	變量兩倍多 = 雙號(蘋果的數 );
	打印線(“後:” + 兩倍多);
	
	變量y = 11;
	打印線(雙號(y));
}

功能雙號(x){
	變量翻倍X = 2 * x;
