/* This program prompts the user for their name
 * and age, and then prints it back out. */
function start(){
	var name = readLine("Name? ");
	println("Name is " + name);
	
	var age = readInt("Age? ");
	println("Age is " + age);
}
		
/*這個程序會提示用戶輸入他們的名字 
*和年齡，然後打印它退了出去。 */ 
函數的開始（）{
	變量名=讀線（“姓名？”）; 
	打印行（“名稱為”+名稱）; 

	可變年齡=讀整型（“年齡？”）; 
	打印線（“年齡”+歲）; 
}
